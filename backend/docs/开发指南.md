import logger from '../utils/logger';

export const exampleService = {
  /**
   * åˆ›å»ºç¤ºä¾‹
   */
  async create(data: { name: string; description?: string; userId: string }) {
    try {
      const [id] = await db('examples').insert({

        description: data.description,
        user_id: data.userId,
        created_at: new Date(),
        updated_at: new Date()
      });

      logger.info(`[ExampleService] Created example: ${id}`);

      return { id, ...data };
    } catch (error) {
      logger.error('[ExampleService] Create failed:', error);
      throw new Error('Failed to create example');
    }
  },

  /**
   * è·å–åˆ—è¡¨
   */
  async list(params: { userId: string; page: number; limit: number }) {
    const offset = (params.page - 1) * params.limit;

    const [items, [{ total }]] = await Promise.all([
      db('examples')
        .where('user_id', params.userId)
        .orderBy('created_at', 'desc')
        .limit(params.limit)
        .offset(offset),
      db('examples')
        .where('user_id', params.userId)
        .count('* as total')
    ]);

    return {
      items,
      total,
      page: params.page,
      limit: params.limit
    };
  }
};
